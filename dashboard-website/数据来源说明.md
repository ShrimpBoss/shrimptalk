## 🔌 如何对接真实数据

### 方案 1：Shopify API（推荐）

```javascript
// 替换模拟数据为真实 Shopify 数据
async function getShopifyMetrics() {
  const response = await fetch('https://your-store.myshopify.com/admin/api/orders.json', {
    headers: { 'X-Shopify-Access-Token': 'your_token' }
  });
  const data = await response.json();
  return {
    gmv: calculateGMV(data.orders),
    orders: data.orders.length,
    profit: calculateProfit(data.orders)
  };
}
```

### 方案 2：Meta Ads API

```javascript
// 获取真实广告数据
async function getMetaAdsData() {
  const response = await fetch('https://graph.facebook.com/v18.0/act_YOUR_AD_ACCOUNT/insights', {
    headers: { 'Authorization': 'Bearer YOUR_ACCESS_TOKEN' }
  });
  const data = await response.json();
  return {
    adSpend: data.ad_spend,
    roas: data.roas,
    conversions: data.conversions
  };
}
```

### 方案 3：自定义数据库

```javascript
// 从真实业务数据库读取
const realData = db.prepare(`
  SELECT 
    SUM(order_total) as gmv,
    COUNT(*) as orders,
    SUM(profit) as profit
  FROM orders 
  WHERE date = ?
`).get(today);
```
